\begin{slide}{Numeradas}
  Equation:
  \begin{equation}
    \dot{x} = Ax+Bu
  \end{equation}

  \vspace*{\fill} % Adiciona espaço de forma que tudo fique igualmente espaçado

  Align:
  \begin{align}
    \dot{x} & = Ax+Bu \\
    y       & =Cx+Du
  \end{align}

  Equation+Aligned:
  \begin{equation}
    \begin{aligned}
      \dot{x} & = Ax+Bu \\

    \end{aligned}
  \end{equation}
